        function createPidTrackingCookie(name,value,days) {
            // This cookie needs to expire after 3 minutes, so pass in 1 for 'days'

                var date = new Date();
                date.setTime(date.getTime()+(days*6000*30));
                var expires = "; expires="+date.toGMTString();
