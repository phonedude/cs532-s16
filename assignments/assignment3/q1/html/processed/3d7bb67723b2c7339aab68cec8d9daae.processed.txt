            $(function () {

                // Title
                var btcp = $('#ticker_price').html();
                $('title').prepend('(' + btcp + ') ');

                if (typeof io != 'undefined') {
                    var socket = io.connect('https://ws.bitcoin.de', {port: 443});

                    socket_update_counter_func = function () {
                        secs_since_last_change = secs_since_last_change + 5;

                        if (50 <= secs_since_last_change) {
                            resetUpdateCounter();

                            return;
                        }

                        $('#last_update_before').text(secs_since_last_change);
                    };

                    if (typeof socket != 'undefined') {
                        socket.on('connect', function () {
                            resetUpdateCounter();
                            socket_update_counter = window.setInterval(socket_update_counter_func, 5000);
                        });

                        socket.on('disconnect', function () {
                            $('#last_update_before').text('--');
                            window.clearInterval(socket_update_counter);
                        });

                        socket.on('refresh_express_option', function (trades) {
                            resetUpdateCounter();
                            refreshExpressOption(trades);
                        });

                        socket.on('remove_order', function (trade) {
                            resetUpdateCounter();
                            $('#trade_id_' + trade.id).toggle('highlight', 500, function () {

                                $(this).remove();
                                var $trades = $('#box_buy_sell_' + trade.type + ' tr');

                                                                if (6 >
                                $trades.filter(':visible').length
                                )
                                {
                                    $trades.filter(':hidden:first').removeClass('dnone').effect('highlight', 2000);
                                }

                                if (0 === $trades.length) {
                                    $.ajax({
                                        type: "GET",
                                        url: "/de/home/reload-trades",
                                        data: {'type': trade.type},
                                        dataType: 'html'
                                    }).done(function (msg) {
                                        $('#box_buy_sell_' + trade.type).empty().append(msg);
                                    });
                                }
                            });
                        });

                        socket.on('add_order', function (trade) {
                            if (1 === $('#trade_id_' + trade.id).length) {
                                return;
                            }

                                                        if (1 === parseInt(trade.is_shorting)) {
                                return;
                            }
                                                        if (1 === parseInt(trade.is_shorting_allowed)) {
                                return;
                            }

                                                        var $tr = getResultsRow(trade);
                            $tr.addClass('dnone');

                            $box_trades = $('#box_buy_sell_' + trade.type);
                            $box_trades.append($tr);

                                                        sort_fields = {};
                            if ('offer' == trade.type) {
                                sort_fields['critical_price'] = 'ASC';
                            }
                            else {
                                sort_fields['critical_price'] = 'DESC';
                            }

                                                        var sorted = $box_trades.find('tr').toArray().sort(function (a, b) {
                                return tradeSorter(a, b, sort_fields);
                            });

                            $box_trades.empty();

                                                        $.each(sorted, function (index, value) {
                                $box_trades.append(value);
                            });

                            var $trades = $box_trades.find('tr');
                            var last_price = parseFloat($trades.filter(':visible:last').attr('data-critical-price'));

                            if (('order' === trade.type && last_price < parseFloat(trade.price))
                                || ('offer' === trade.type && last_price > parseFloat(trade.price))) {
                                resetUpdateCounter();
                                $tr.removeClass('dnone').effect('highlight', 2000);
                                $trades.filter(':visible:gt(5)').addClass('dnone');
                            }

                            $trades.filter(':hidden:gt(19)').remove();
                        });

                        socket.on('browser_reload', function () {
                            window.location.reload();
                        });
                    }
                }
            });

            function resetUpdateCounter() {
                secs_since_last_change = 0;
                $('#last_update_before').text(secs_since_last_change);
                $('#last_update_before_wrap strong').effect('highlight', 2000);
            }

        

    function getResultsRow(trade) {
        var current_short_culture = 'de';
        var tr = $('<tr />', {
            'data-trade-id': trade.id,
            'id': 'trade_id_' + trade.id,
            'data-critical-price-formatted': trade['price_formatted_' + current_short_culture],
            'data-critical-price': trade.price,
            'data-amount': trade.amount,
            'data-trade-type': trade.type
        });

        var link_offer = '<a class="c6" href="/de/buyOffer?trade_id=abc">KAUFEN</a>';
        var link_order = '<a class="c6" href="/de/sellOrder?trade_id=abc">VERKAUFEN</a>';
        var $no_link = $('<span>---</span>');

        if ('offer' === trade.type) {
            var link = link_offer.replace('abc', trade.id);
        }
        else {
            var link = link_order.replace('abc', trade.id);
        }

        
        var payment_method_div = $('<div/>', {'class': 'prelative'});
        var alt = "Sitz der Bank: %land%";
        alt = alt.replace('%land%', trade['country_payment_method_' + current_short_culture]);
        var payment_method_img = $('<img/>', {
            'src': '/images/currency_icon/' + trade.seat_of_bank_of_creator + '.png',
            'alt': alt,
            'title': alt
        });

        $('<td/>', {
            'data-cell': 'amount',
            'text': trade['amount_' + current_short_culture] + ' (' + (trade['min_amount_' + current_short_culture]) + ')'
        }).appendTo(tr);
        $('<td/>', {'class': 'aright', 'data-cell': 'price', 'text': trade['price_' + current_short_culture]}).appendTo(tr);
        $('<td/>', {
            'class': 'aright',
            'data-cell': 'volume',
            'text': trade['volume_' + current_short_culture]
        }).appendTo(tr);
        $('<td/>', {'class': 'hidden-xs', 'data-cell': 'payment_methods', html: payment_method_div.append(payment_method_img)}).appendTo(tr);
        $('<td/>', {'class': 'p0 hidden-xs', 'data-cell': 'not_used'}).appendTo(tr);
        $('<td/>', {'class': 'w60 hidden-xs', 'data-cell': 'additional_info'}).appendTo(tr);
        $('<td/>', {'data-cell': 'buy_sell_action', html: link}).appendTo(tr);

                if (1 === parseInt(trade.is_trade_by_fidor_reservation_allowed) && 0 === parseInt(trade.is_trade_by_sepa_allowed)) {
            tr.find('td[data-cell=additional_info]').append('<img class="mbm2" data-express="express" title="Express-Handel" alt="Express-Handel" src="/images/icons/fidor.png" />');
        }

        if (1 === parseInt(trade.is_trade_by_fidor_reservation_allowed) && 1 === parseInt(trade.is_trade_by_sepa_allowed)) {
            tr.find('td[data-cell=additional_info]').append('<img class="mbm2" data-express="express_and_sepa" title="Express-Handel oder Standard-Überweisung" alt="Express-Handel oder Standard-Überweisung" src="/images/icons/fidor_sepa.png" />');
        }
        
        if ('1' == trade.is_shorting) {
            tr.find('td[data-cell=additional_info]').append('<img class="mbm2" title="Verkäufer hat sich nicht verpflichtet, die zu verkaufenden Bitcoins vor dem Abschluss des Kaufvertrages einzuliefern." alt="Verkäufer hat sich nicht verpflichtet, die zu verkaufenden Bitcoins vor dem Abschluss des Kaufvertrages einzuliefern." src="/images/icons/no_btc.jpg" />');
        }
        if ('1' == trade.is_shorting_allowed) {
            tr.find('td[data-cell=additional_info]').append('<img class="mbm2" title="Leerverkauf" alt="Leerverkauf" src="/images/icons/no_btc.jpg" />');
        }

        if ('1' == trade.is_kyc_full) {
            tr.find('td[data-cell=additional_info]').append('&nbsp;<img class="mbm2" title="Vollständig identifizierter User" alt="Vollständig identifizierter User" src="/images/icons/verified.png" />');
        }

        return tr;
    }

    function refreshExpressOption(trades) {
        $.each(trades, function (trade_id, trade_info) {
            $icon = $('#trade_id_' + trade_id + ' img[data-express]');
            var express_allowed = parseInt(trade_info.is_trade_by_fidor_reservation_allowed);
            var sepa_allowed = parseInt(trade_info.is_trade_by_sepa_allowed);

            if (0 === express_allowed && 1 === parseInt($icon.length)) {
                $icon.remove();
            }

            
            $trade_row = $('#trade_id_' + trade_id);

            if (true === $trade_row.is(':visible')) {
                $trade_row.effect('highlight', 2000);
            }
        });
    }
