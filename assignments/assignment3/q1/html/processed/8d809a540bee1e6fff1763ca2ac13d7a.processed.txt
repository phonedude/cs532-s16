	function extractParamFromUri(uri, paramName) {
		if (!uri) {
			return;
		}
		var regex = new RegExp('[\\?&#]' + paramName + '=([^&#]*)');
		var params = regex.exec(uri);
		if (params != null) {
			return unescape(params[1]);
		}
		return;
