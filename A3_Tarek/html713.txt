	function check_form() {
		var name = $('#msg_name');

            alert('Не введен код с картинки.');
            captcha.focus();
            return false;

		if ( name_val.length < 1 ) {
			alert('Не заполнено поле имя');
            name.focus();
			return false;
		}

			alert('Имя отправителя не должно превышать 50 символов');

			return false;

			alert('Длина адреса электронной почты превышает 50 символов');
			email.focus();
            return false;
		}
		if  (email_val.length && !match ) {
			alert('Адрес электронной почты указан некорректно');
            email.focus();
			return false;

		return true;
