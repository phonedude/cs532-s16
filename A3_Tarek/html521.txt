function addLink() {
            if (document.location.href.indexOf("/adm/") >= 0) return;
            var copy = "Смотрите оригинал материала на <a href=\"" + window.location.href + "\">" + window.location.href + "</a>";
            var hdn = document.createElement('span');
            hdn.id = "art_copy";
            hdn.innerHTML = "<br /><br />" + copy;
            if (document.selection) {
                var sel = document.selection;
                var rng = sel.createRange();
                var txt = rng.text;
                if (txt.indexOf("1tv.ru") == -1) {
                    txt = rng.duplicate();
                    txt.collapse(false);
                    txt.pasteHTML(hdn.outerHTML);
                    rng.setEndPoint("EndToEnd", txt);
                    rng.select();
                }
            } else {
                if (document.getSelection) {
                    var sel = window.getSelection();
                    var txt = sel.toString();
                    if (txt.indexOf("1tv.ru") == -1) {
                        var rng = sel.getRangeAt(0);
                        txt = rng.cloneRange();
                        txt.collapse(false);
                        txt.insertNode(hdn);
                        rng.setEndAfter(hdn);
                        sel.removeAllRanges();
                        sel.addRange(rng);
                    }
                }
            }
        }
function CreateCopyrightHandler() {
            if (document.getElementById("article_copy")) {
                document.getElementById("article_copy").oncopy = addLink;
            }
        }
        window.setTimeout('CreateCopyrightHandler()', 300);

function checkEmailAddress(email_str) {
  var goodEmail = email_str.match(/\b(^(\S+@).+((\.com)|(\.net)|(\.edu)|(\.mil)|(\.gov)|(\.org)|(\.info)|(\.sex)|(\.biz)|(\.aero)|(\.coop)|(\.museum)|(\.name)|(\.pro)|(\..{2,2}))$)\b/gi);
  if (goodEmail) return true;
  else return false;
}

function eMailerResponse(responseText, statusText)  {
 if (statusText == "success") {
   if (responseText.search(/^NOCLEAR\./i) == -1) {
     $("#eMailer *:not(.nonClearFields)").clearFields();
   } else {
     responseText = responseText.replace(/^NOCLEAR\./, '');
   }
   $("#eMailerResult").empty().prepend(responseText).eMailerFormWait();
 }
 else {
   $("#eMailerResult").empty().prepend("Произошла ошибка. Попробуйте отправить сообщение ещё раз.");
 }
}

    time = time || 5000;

      setTimeout(function() {
        $(self).dequeue();
